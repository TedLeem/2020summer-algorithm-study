``` python
#스택-괄호#

T = int(input())
result = []
stack_list = []

for _ in range(T):
    stack = input()

    for s in stack:
        if s == '(':
            stack_list.append(s)

        elif s == ')':
            if len(stack_list) == 0:
                stack_list.append(s)
                break

            for i in stack_list:
                if i == '(':
                    stack_list.remove(i)
                    break

    if len(stack_list) == 0:
        result.append("YES")

    elif len(stack_list) != 0:
        result.append("NO")
    stack_list.clear()

for r in result:
    print(r)
```
