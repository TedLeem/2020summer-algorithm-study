def solution(word, pages):
    answer = 0
    word = word.lower()
    
    url_index = {}
    exlink_cnt = {}
    exlink_score = {}
    matching_score = {}

    for i in range(len(pages)):
        page = pages[i].lower()
        url = re.search(r'<meta[^>]*content="https://([\S]*)"/>', page).group(1)
        url_index[url] = i
        
        word_cnt = 0
        for find in re.findall(r'[a-zA-Z]+', page):
            if find == word:
                word_cnt += 1
                
    return answer
